<!DOCTYPE html>
<html>
<head>
    <style>
        body {
            margin: 0;
            overflow: hidden;
            background-color: transparent;
        }
        .name {
            position: absolute;
            font-size: 20px;
            animation: bounce 5s infinite;
        }
        @keyframes bounce {
            0%, 100% {
                transform: translateY(0);
            }
            50% {
                transform: translateY(-30px);
            }
        }
        #input-container {
            position: absolute;
            top: 10px;
            left: 10px;
            z-index: 10;
            background: rgba(0, 0, 0, 0.5); /* Semi-transparent background for input container */
            padding: 10px;
            border-radius: 5px;
        }
    </style>
    <script>
        // Function to generate a random color
        function getRandomColor() {
            const letters = '0123456789ABCDEF';
            let color = '#';
            for (let i = 0; i < 6; i++) {
                color += letters[Math.floor(Math.random() * 16)];
            }
            return color;
        }

        // Function to add a new name
        function addName() {
            const nameInput = document.getElementById('name-input');
            const name = nameInput.value;
            if (name) {
                const nameDiv = document.createElement('div');
                nameDiv.className = 'name';
                nameDiv.style.color = getRandomColor();
                nameDiv.style.top = Math.random() * 90 + '%'; // Random position
                nameDiv.style.left = Math.random() * 90 + '%'; // Random position
                nameDiv.textContent = name;
                document.body.appendChild(nameDiv);
                nameInput.value = '';
            }
        }
    </script>
</head>
<body>
    <div id="input-container">
        <input type="text" id="name-input" placeholder="Enter subscriber name">
        <button onclick="addName()">Add Name</button>
    </div>
</body>
</html>
